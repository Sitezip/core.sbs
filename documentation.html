Quick Start

1 Define a Template 

<section id="cr-data" style="display:none;">
    <template name="userCard">
        <div class="user-card">
            <h3>{{rec:name}}</h3>
            <p>{{rec:email}}</p>
            <img {{rec:avatar:core_pk_attr:src}} alt="User Avatar">
        </div>
    </template>
</section>

2 Create a Pocket 

<div class="core-pocket" 
     data-core-templates="userCard" 
     data-core-source-userCard="https://api.example.com/users/1">
</div>

Data Referencing

Syntax:

{{type:object-member:format:clue}}

Examples

Syntax Description
{{rec:name}} Simple property access
{{rec:balance:money:$}} Currency formatting with dollar sign
{{rec:status:upper}} Uppercase transformation
{{rec:email:emaillink}} Email link generation
{{rec:url:urllink}} URL link generation
{{rec:image:imgsrc}} Image source attribute
{{rec:customer.address.city}} Nested property access
{{rec:items:core_pk_cloner:item}} Array cloning with template

Attributes

Attribute Description Example
data-core-templates Template names to load data-core-templates="user,profile"
data-core-source Default data source URL data-core-source="/api/user"
data-core-source-[name] Template-specific source data-core-source-profile="/api/profile"

Custom Functions

Initialization

// Initialize Core.js
core.ud.init = () => {
    core.useRouting = false;
    core.useDebugger = window.location.hostname === 'localhost';
};

Lifecycle Hooks

// Before data fetch
core.ud.preflight = (dataRef, dataSrc, type) => {
    // Modify request settings
    return { dataRef, dataSrc, type };
};

// After data fetch, before rendering
core.ud.postflight = (dataRef, dataObj, type) => {
    // Process data before rendering
    return dataObj;
};

Custom Formatters

core.ud.formatValue = (value, formatList, clue) => {
    for (const format of formatList) {
        switch (format) {
            case 'uppercase': return String(value).toUpperCase();
            case 'lowercase': return String(value).toLowerCase();
            case 'truncate': 
                const length = clue ? parseInt(clue) : 100;
                return value.length > length ? value.substring(0, length) + '...' : value;
        }
    }
    return value;
};

API Reference

Core Methods

Method Description Example
core.cr.setData(name, data) Store data in registry core.cr.setData('user', {name: 'John'})
core.cr.getData(name) Retrieve data from registry const user = core.cr.getData('user')
core.be.getData(ref, src) Fetch data from URL core.be.getData('users', '/api/users')
